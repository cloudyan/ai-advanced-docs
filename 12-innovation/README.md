#### 问题评估框架

```python
"""
问题价值评估（SCORE 框架）：

S - Significance（重要性）
  - 影响多少人/企业？
  - 经济价值多大？
  - 学术意义如何？

C - Clarity（清晰度）
  - 问题能否明确定义？
  - 成功标准是否清晰？
  - 范围是否可控？

O - Originality（原创性）
  - 是否有人做过？
  - 有什么独特之处？
  - 创新性如何？

R - Resources（资源可行性）
  - 需要多少数据？
  - 需要多少算力？
  - 需要多长时间？

E - Execution（执行可行性）
  - 团队能力是否匹配？
  - 技术风险是否可控？
  - 是否有备选方案？

评分标准：
每个维度 1-5 分，总分 25 分
- 20-25 分：值得投入
- 15-19 分：可以考虑
- 10-14 分：需要重新定义
- <10 分：放弃或大幅修改
"""

def evaluate_problem(problem_description):
    """评估研究问题（示例模板）"""
    criteria = {
        "Significance": {
            "questions": [
                "这个问题影响多少人？",
                "经济价值多大？",
                "学术意义如何？"
            ],
            "score": 0  # 1-5 分
        },
        "Clarity": {
            "questions": [
                "问题能否明确定义？",
                "成功标准是否清晰？",
                "范围是否可控？"
            ],
            "score": 0
        },
        "Originality": {
            "questions": [
                "是否有人做过？",
                "有什么独特之处？",
                "创新性如何？"
            ],
            "score": 0
        },
        "Resources": {
            "questions": [
                "需要多少数据？",
                "需要多少算力？",
                "需要多长时间？"
            ],
            "score": 0
        },
        "Execution": {
            "questions": [
                "团队能力是否匹配？",
                "技术风险是否可控？",
                "是否有备选方案？"
            ],
            "score": 0
        }
    }
    
    # 打印评估问题
    for criterion, info in criteria.items():
        print(f"\n{criterion}:")
        for q in info["questions"]:
            print(f"  - {q}")
        score = int(input(f"  评分 (1-5): "))
        info["score"] = score
    
    # 计算总分
    total = sum(info["score"] for info in criteria.values())
    print(f"\n{'='*40}")
    print(f"总分：{total}/25")
    
    if total >= 20:
        print("评估：值得投入！")
    elif total >= 15:
        print("评估：可以考虑")
    elif total >= 10:
        print("评估：需要重新定义")
    else:
        print("评估：建议放弃")
    
    return total
```
